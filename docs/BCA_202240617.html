The manual for the Pierce BCA Protein Assay suggest that a linear regression might be worse than non-linear regression for the calibration curve. It suggests to use a ‘four-parameter quadratic’ equation, which is commonly called a logistic regression. Let’s use it here:

source("R/calibrate_assay.R")

standards <- filter(data, task == "Standard")
calibration_curve <- fit_logistic_calibration_curve(standards, level = 0.95)

plot1 <- ggplot(data = standards, aes(x = quantity, y = a560)) +
  geom_point() +
  geom_ribbon(
    data = calibration_curve$fit,
    mapping = aes(x = quantity, ymin = lwr, ymax = upr),
    fill = "dodgerblue",
    alpha = 0.2,
    inherit.aes = FALSE
  ) +
  geom_line(
    data = calibration_curve$fit,
    mapping = aes(x = quantity, y = fit),
    linetype = "dashed",
    inherit.aes = FALSE
  ) +
  annotate(
    geom = "text",
    label = 'paste(psi, {R^2}["adj"], " = ")',
    x = 0,
    y = 2,
    parse = TRUE,
    hjust = "left"
  ) +
  geom_text(
    data = calibration_curve$r_squared,
    mapping = aes(x = 0.28, y = 2, label = round(adj_R_squared, 4))
  ) +
  labs(
    x = "Albumin, mg/mL",
    y = "Absorbance A560",
    title = "Calibration curve",
    subtitle = "Fit + CI(95%)"
  ) +
  theme_prism()
